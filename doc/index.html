# how to use it?

{% highlight scss %} .button { @include button( $font-color, $background-color, $border-radius, $font-weight, $boolean-button-ghost, $boolean-uppercase ); } {% endhighlight %}
Button {% highlight scss %} .button { @include button(#fff, #8cd1a8); } {% endhighlight %}
Ghost {% highlight scss %} .btn-ghost { @include button(#fff, #8cd1a8, 0, 300, capitalize, true); } {% endhighlight %}
Ghost primary {% highlight scss %} .btn-ghost-primary { @include button(#8cd1a8, #8cd1a8, 0, 700, capitalize, true); } {% endhighlight %}
uppercase lowercase {% highlight scss %} .btn-ghost-upper { @include button(#fff, #8cd1a8, 0, 500, uppercase, true); } .btn-ghost-lower { @include button(#fff, #8cd1a8, 0, 500, lowercase, true); } {% endhighlight %}
Button Blue {% highlight scss %} .btn-blue { @include button(#fff, #4696e5, 0, 700); } {% endhighlight %}
Ghost Blue {% highlight scss %} .btn-ghost-blue { @include button(#4696e5, #8cd1a8, 0, 500, capitalize, true); } {% endhighlight %}
